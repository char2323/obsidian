### 408 排序算法全解析（带详细注释版）

```cpp
#include <algorithm>
#include <iostream>
#include <source_location>
#include <utility>
#include <vector>

using namespace std;

// 利用 C++20 特性辅助调试，自动打印当前调用的函数名
void log_call(const std::source_location location = std::source_location::current()) {
    std::cout << "\n>>> 正在执行: " << location.function_name() << " <<<" << std::endl;
}

void showarr(const vector<int> &arr);
void showarr2(const vector<int> &arr);

/* ================== 1. 选择排序 (Selection Sort) ================== */
// 特点：每一趟确定一个最小值的最终位置。
// 408必考：不稳定性（如 [5, 5, 2] 第一趟后两个5顺序反了）、比较次数与初始状态无关。
void selectSort(vector<int> &arr) {
    log_call();
    int n = arr.size();
    for (int i = 0; i < n - 1; ++i) {
        int min = i; 
        for (int j = i + 1; j < n; ++j) {
            if (arr[j] < arr[min]) min = j; // 记录本趟最小值的下标
        }
        if (min != i) swap(arr[i], arr[min]); // 交换到已排序序列的末尾
    }
    showarr(arr);
}

/* ================== 2. 冒泡排序 (Bubble Sort) ================== */
// 特点：每一趟“浮”出一个最大/小值。
// 408必考：稳定性、最好情况 O(n)（基本有序时通过 swapped 提前结束）。
void bubbleSort(vector<int> &arr) {
    log_call();
    int n = arr.size();
    for (int i = 0; i < n - 1; i++) {
        bool swapped = false; // 优化：若本趟无交换，说明已有序
        for (int j = n - 1; j > i; j--) { // 从后往前冒泡，小的往上浮
            if (arr[j - 1] > arr[j]) {
                swap(arr[j - 1], arr[j]);
                swapped = true;
            }
        }
        if (!swapped) break;
    }
    showarr(arr);
}

/* ================== 3. 插入排序 (Insertion Sort) ================== */
// 特点：将未排序元素插入到前面已排序的序列中。
// 408必考：稳定性、基本有序时效率最高。

// 3.1 直接插入 (从0开始，不带哨兵)
void insertSort1(vector<int> &arr) {
    log_call();
    int n = arr.size();
    for (int i = 1; i < n; i++) {
        if (arr[i] < arr[i - 1]) { // 发现逆序，才需移动
            int temp = arr[i], j;
            // j >= 0 防止越界，temp < arr[j] 找位置的同时后移元素
            for (j = i - 1; j >= 0 && temp < arr[j]; j--) {
                arr[j + 1] = arr[j];
            }
            arr[j + 1] = temp; // 插入到合适位置
        }
    }
    showarr(arr);
}

// 3.2 直接插入 (从1开始，arr[0]作哨兵)
void insertSort4(vector<int> &arr) {
    log_call();
    int n = arr.size();
    int j, i;
    for (i = 2; i < n; i++) { // 下标1默认有序，从2开始
        if (arr[i - 1] > arr[i]) {
            arr[0] = arr[i]; // 哨兵：保存待插入元素，并充当边界边界挡箭牌
            for (j = i - 1; arr[j] > arr[0]; j--) { // 不需要判断 j>=0
                arr[j + 1] = arr[j];
            }
            arr[j + 1] = arr[0]; // 哨兵落位
        }
    }
    showarr2(arr);
}

// 3.3 折半插入 (Binary Insertion Sort)
// 特点：利用二分查找减少“比较次数”，但“移动次数”没有减少。
void binsertSort(vector<int> &arr) {
    log_call();
    for (int i = 1; i < arr.size(); i++) {
        int temp = arr[i];
        int low = 0, high = i - 1;
        while (low <= high) { // 二分查找待插入位置
            int mid = low + (high - low) / 2;
            if (arr[mid] > temp) high = mid - 1; // 插入点在左半区
            else low = mid + 1;                  // 插入点在右半区
        }
        // 移动 low 之后的所有元素
        for (int j = i - 1; j >= low; j--) arr[j + 1] = arr[j];
        arr[low] = temp;
    }
    showarr(arr);
}

/* ================== 4. 希尔排序 (Shell Sort) ================== */
// 特点：缩小增量排序，本质是分组插入。
// 408必考：不稳定性。
void shellSort(vector<int> &arr) {
    log_call();
    int n = arr.size();
    for (int d = n / 2; d >= 1; d /= 2) { // 增量序列
        for (int i = d; i < n; i++) {    // 对各组进行直接插入排序
            if (arr[i] < arr[i - d]) {
                int temp = arr[i], j;
                for (j = i - d; j >= 0 && arr[j] > temp; j -= d) {
                    arr[j + d] = arr[j];
                }
                arr[j + d] = temp;
            }
        }
    }
    showarr(arr);
}

/* ================== 5. 快速排序 (Quick Sort) ================== */
// 特点：分治法。
// 408必考：最坏情况 O(n^2)（已有序时）、平均性能最优、空间 O(logn)。

int Partition(vector<int> &arr, int low, int high) {
    int pivot = arr[low]; // 第一个元素作为枢轴
    while (low < high) {
        while (low < high && arr[high] >= pivot) high--; // 从右往左找小的
        arr[low] = arr[high]; // 填左坑
        while (low < high && arr[low] <= pivot) low++;   // 从左往右找大的
        arr[high] = arr[low]; // 填右坑
    }
    arr[low] = pivot; // 枢轴最终归位
    return low;
}

void quickSort(vector<int> &arr, int low, int high) {
    if (low < high) {
        int pivotPos = Partition(arr, low, high);
        quickSort(arr, low, pivotPos - 1);
        quickSort(arr, pivotPos + 1, high);
    }
}

/* ================== 6. 堆排序 (Heap Sort) ================== */
// 特点：利用大/小根堆特性进行排序。
// 408必考：不稳定性、建堆复杂度 O(n)、排序总复杂度 O(nlogn)。

// 向下调整函数 (以大根堆为例)
void MaxHeapAdjust(vector<int> &arr, int k, int len) {
    int temp = arr[k]; // 暂存根节点
    for (int i = 2 * k; i <= len; i *= 2) { // i 为左孩子下标
        if (i < len && arr[i] < arr[i + 1]) i++; // 选取较大的孩子
        if (temp >= arr[i]) break;               // 父节点已比孩子大，调整结束
        arr[k] = arr[i]; // 将大孩子上移
        k = i;           // 更新坑位下标，继续向下筛选
    }
    arr[k] = temp; // 将最初的根节点放入最终确定的“坑”
}

void maxheapSort(vector<int> &arr) {
    log_call();
    int n = arr.size() - 1; // 假设有效数据从 1 开始
    // 1. 建立初始大根堆：从最后一个非叶子节点 (n/2) 开始向下调整
    for (int i = n / 2; i > 0; i--) MaxHeapAdjust(arr, i, n);

    // 2. 排序：不断交换堆顶与堆底，并缩小堆范围
    for (int i = n; i > 1; i--) {
        swap(arr[i], arr[1]);       // 最大的放到最后
        MaxHeapAdjust(arr, 1, i - 1); // 对剩余堆进行调整
    }
    showarr2(arr);
}

/* ================== 7. 归并排序 (Merge Sort) ================== */
// 特点：二路归并，分而治之。
// 408必考：空间复杂度 O(n)、稳定性、比较次数与初始状态无关。

void merge(vector<int> &A, int low, int mid, int high) {
    vector<int> B = A; // 深拷贝辅助数组 (408标准写法)
    int i = low, j = mid + 1, k = low;
    // 比较 B 中的两段，结果写入 A
    while (i <= mid && j <= high) {
        if (B[i] <= B[j]) A[k++] = B[i++]; // 加等号保证稳定性
        else A[k++] = B[j++];
    }
    // 处理剩余部分
    while (i <= mid) A[k++] = B[i++];
    while (j <= high) A[k++] = B[j++];
}

void mergeSort(vector<int> &arr, int low, int high) {
    if (low < high) {
        int mid = low + (high - low) / 2;
        mergeSort(arr, low, mid);       // 左子序列递归
        mergeSort(arr, mid + 1, high);  // 右子序列递归
        merge(arr, low, mid, high);     // 合并
    }
}
```

### 一、排序“趟数”与初始状态的关系

**结论：** 只有 **交换类排序**（冒泡、快排）的趟数与初始状态有关。

|**算法类型**|**趟数是否与初始状态有关**|**原因分析**|
|---|---|---|
|**直接插入排序**|**无关**|始终需要 $n-1$ 趟处理（即使数组有序，也要走完循环）。|
|**简单选择排序**|**无关**|始终需要 $n-1$ 趟，每趟选出一个最小值。|
|**冒泡排序**|**有关**|若一趟中没有发生交换（优化版），则提前结束。|
|**快速排序**|**有关**|趟数取决于递归树的高度。初始基本有序时，树最深，趟数最多（$n-1$ 趟）；平衡时最少（$\log_2 n$ 趟）。|
|**堆排序**|**无关**|调整过程是固定的，总趟数为 $n-1$。|
|**归并排序**|**无关**|始终需要 $\lceil \log_2 n \rceil$ 趟归并。|

---

### 二、 “比较次数”与初始状态的关系

**结论：** 只有**简单选择排序**和**归并排序**的比较次数与初始状态**无关**。

|**算法类型**|**比较次数是否有关**|**细节说明**|
|---|---|---|
|**直接插入排序**|**有关**|最好情况（正序）比较 $1$ 次即可；最坏情况（逆序）比较 $i$ 次。|
|**折半插入排序**|**无关***|**注意！** 408 认为折半插入的**比较次数**与初始状态无关（因为二分查找的次数固定），但总时间复杂度仍有关。|
|**简单选择排序**|**无关**|无论如何都要扫描剩余序列找最小值，比较次数始终是 $n(n-1)/2$。|
|**冒泡排序**|**有关**|初始有序时比较一次发现没交换就结束；逆序时每一对都要比。|
|**快速排序**|**有关**|枢轴的选择决定了比较次数。越不平衡（已有序），比较次数越多。|
|**堆排序**|**有关**|虽然数量级是 $O(n \log n)$，但具体比较会受节点值大小影响而提前停止（`temp >= arr[i]`）。|
|**归并排序**|**无关**|每一趟合并的比较次数是固定的 $O(n)$ 级别。|

---

### 三、 “移动次数”与初始状态的关系

**结论：** 绝大多数算法的移动次数都与初始状态有关。

- **唯一例外：简单选择排序**。虽然它的比较次数很多，但它的**移动次数最少**。每一趟最多交换一次（$3$ 次赋值），移动次数在 $O(n)$ 数量级。
    
- **快速排序、插入排序、冒泡排序**：移动（或交换）次数严重依赖初始逆序程度。
    

---

### 四、 408 考场高频坑点总结（必背）

1. **“不长记性”的算法**：**简单选择排序**。它很“笨”，不管数组多有序，它都要老老实实地比较完 $n(n-1)/2$ 次，只有交换的时候才稍微省点力气。
    
2. **“最怕遇到对的人”**：**快速排序**。当数组已经排好序（正序或逆序）时，快排表现最差，递归树退化为单支树，效率降为 $O(n^2)$。
    
3. **“性能最稳”**：**归并排序**。无论数组长什么样，它的比较次数、趟数、时间复杂度、稳定性都雷打不动，代价是 $O(n)$ 的辅助空间。
    
4. **折半插入的特殊性**：二分查找优化了比较次数（变成 $O(n \log n)$），但没有优化移动次数（仍是 $O(n^2)$）。
